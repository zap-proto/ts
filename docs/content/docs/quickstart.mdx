This guide walks you through connecting to a ZAP gateway and calling tools.

## Browser Example

ZAP works in browsers using WebSocket transport.

```typescript
import { Client } from '@hanzo/zap';

async function main() {
  // Connect via WebSocket
  const client = await Client.connect('wss://gateway.example.com:9999');

  try {
    // Discover available tools
    const tools = await client.listTools();
    console.log('Available tools:', tools.map(t => t.name));

    // Call a tool
    const result = await client.callTool('search', {
      query: 'TypeScript documentation',
    });
    console.log('Search result:', result);

    // List resources
    const resources = await client.listResources();
    console.log('Resources:', resources.map(r => r.name));

    // Read a resource
    if (resources.length > 0) {
      const content = await client.readResource(resources[0].uri);
      console.log('Content:', content);
    }
  } finally {
    await client.close();
  }
}

main().catch(console.error);
```

## Node.js Example

In Node.js, you can use TCP or Unix socket transports for better performance.

```typescript
import { Client, Gateway } from '@hanzo/zap';

async function main() {
  // Option 1: Connect to existing gateway
  const client = await Client.connect('zap://localhost:9999');

  // Option 2: Create your own gateway
  const gateway = new Gateway({
    listen: '0.0.0.0',
    port: 9999,
    servers: [
      {
        name: 'filesystem',
        url: 'stdio://npx',
        transport: 'stdio',
        command: 'npx',
        args: ['-y', '@modelcontextprotocol/server-filesystem', '/tmp'],
      },
      {
        name: 'fetch',
        url: 'http://localhost:3001',
        transport: 'http',
      },
    ],
  });

  await gateway.start();
  console.log('Gateway running on port 9999');

  // Connect client to local gateway
  const localClient = await Client.connect('zap://localhost:9999');
  const tools = await localClient.listTools();
  console.log('Aggregated tools:', tools.length);

  await localClient.close();
  await gateway.stop();
}

main().catch(console.error);
```

## Running a Server

Expose your own tools and resources as an MCP server.

```typescript
import { Server } from '@hanzo/zap';

const server = new Server({
  listen: '0.0.0.0',
  port: 9999,
});

// Server runs indefinitely
await server.run();
```

## Configuration via Environment

ZAP reads configuration from environment variables:

```bash
export ZAP_LISTEN=0.0.0.0
export ZAP_PORT=9999
export ZAP_LOG_LEVEL=info
export ZAP_TLS_CERT=/path/to/cert.pem
export ZAP_TLS_KEY=/path/to/key.pem
```

```typescript
import { loadConfigFromEnv, mergeConfig, DEFAULT_CONFIG } from '@hanzo/zap';

const envConfig = loadConfigFromEnv();
const config = mergeConfig(DEFAULT_CONFIG, envConfig, {
  maxConnections: 500,
});
```

## Error Handling

ZAP uses typed errors for precise error handling.

```typescript
import {
  Client,
  ZapError,
  ConnectionError,
  ToolNotFoundError,
  TimeoutError,
} from '@hanzo/zap';

try {
  const client = await Client.connect('zap://localhost:9999');
  await client.callTool('missing-tool', {});
} catch (error) {
  if (error instanceof ToolNotFoundError) {
    console.error('Tool does not exist:', error.details?.toolName);
  } else if (error instanceof ConnectionError) {
    console.error('Connection failed:', error.message);
  } else if (error instanceof TimeoutError) {
    console.error('Request timed out');
  } else if (error instanceof ZapError) {
    console.error('ZAP error:', error.code, error.message);
  }
}
```
