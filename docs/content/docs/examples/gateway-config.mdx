## Development Gateway

```typescript
import { Gateway } from '@hanzo/zap';

const gateway = new Gateway({
  listen: '127.0.0.1',  // Local only
  port: 9999,
  logLevel: 'debug',
  servers: [
    // Filesystem access
    {
      name: 'filesystem',
      url: 'stdio://filesystem',
      transport: 'stdio',
      command: 'npx',
      args: ['-y', '@modelcontextprotocol/server-filesystem', process.cwd()],
    },
    // Memory/scratch pad
    {
      name: 'memory',
      url: 'stdio://memory',
      transport: 'stdio',
      command: 'npx',
      args: ['-y', '@modelcontextprotocol/server-memory'],
    },
  ],
});

await gateway.start();
```

## Production Gateway

```typescript
import { Gateway, loadConfigFromEnv, mergeConfig, DEFAULT_CONFIG } from '@hanzo/zap';

const envConfig = loadConfigFromEnv();

const gateway = new Gateway(mergeConfig(DEFAULT_CONFIG, envConfig, {
  listen: '0.0.0.0',
  port: 9999,
  logLevel: 'info',
  tlsCert: '/etc/ssl/certs/zap.crt',
  tlsKey: '/etc/ssl/private/zap.key',
  maxConnections: 5000,
  connectionTimeout: 30000,
  requestTimeout: 120000,
  servers: [
    // Production MCP servers
    {
      name: 'search',
      url: 'http://search-mcp:3001',
      transport: 'http',
    },
    {
      name: 'database',
      url: 'unix:///var/run/mcp/database.sock',
      transport: 'unix',
    },
    {
      name: 'api',
      url: 'wss://api-mcp.internal:8080',
      transport: 'websocket',
    },
  ],
}));

// Graceful shutdown
const shutdown = async () => {
  console.log('Shutting down gateway...');
  await gateway.stop();
  process.exit(0);
};

process.on('SIGINT', shutdown);
process.on('SIGTERM', shutdown);

await gateway.start();
console.log(`Gateway running on port ${gateway.getConfig().port}`);
```

## Docker Compose Setup

```yaml
# compose.yml
services:
  zap-gateway:
    build: .
    ports:
      - "9999:9999"
    environment:
      ZAP_LOG_LEVEL: info
      ZAP_MAX_CONNECTIONS: 1000
    volumes:
      - ./config:/etc/zap:ro
      - /var/run/mcp:/var/run/mcp
    depends_on:
      - search-mcp
      - database-mcp
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:9999/health"]
      interval: 30s
      timeout: 10s
      retries: 3

  search-mcp:
    image: ghcr.io/example/search-mcp:latest
    expose:
      - "3001"

  database-mcp:
    image: ghcr.io/example/database-mcp:latest
    volumes:
      - /var/run/mcp:/var/run/mcp
```

## Configuration File

```typescript
// config/gateway.ts
import type { Config, ServerConfig } from '@hanzo/zap';

const servers: ServerConfig[] = [
  {
    name: 'filesystem',
    url: 'stdio://filesystem',
    transport: 'stdio',
    command: 'node',
    args: ['/opt/mcp/filesystem/index.js'],
    env: {
      MCP_ROOT: '/data',
      MCP_READ_ONLY: 'true',
    },
  },
  {
    name: 'web-search',
    url: 'http://search-mcp:3001',
    transport: 'http',
  },
  {
    name: 'code-analysis',
    url: 'unix:///var/run/mcp/code.sock',
    transport: 'unix',
  },
];

export const config: Partial<Config> = {
  listen: process.env.ZAP_LISTEN || '0.0.0.0',
  port: parseInt(process.env.ZAP_PORT || '9999', 10),
  logLevel: (process.env.ZAP_LOG_LEVEL as any) || 'info',
  servers,
  maxConnections: 2000,
  connectionTimeout: 30000,
  requestTimeout: 60000,
};
```

## Dynamic Server Management

```typescript
import { Gateway } from '@hanzo/zap';

const gateway = new Gateway({
  port: 9999,
  servers: [], // Start empty
});

await gateway.start();

// Add servers dynamically based on conditions
if (process.env.ENABLE_SEARCH === 'true') {
  await gateway.connectServer({
    name: 'search',
    url: 'http://localhost:3001',
    transport: 'http',
  });
}

if (process.env.ENABLE_DATABASE === 'true') {
  await gateway.connectServer({
    name: 'database',
    url: 'unix:///var/run/mcp/db.sock',
    transport: 'unix',
  });
}

// Monitor server status
setInterval(() => {
  const servers = gateway.listServers();
  for (const server of servers) {
    if (server.status === 'error') {
      console.error(`Server ${server.name} is in error state`);
      // Attempt reconnection
      gateway.disconnectServer(server.id)
        .then(() => gateway.connectServer({
          name: server.name,
          url: server.url,
          transport: 'http',
        }))
        .catch(console.error);
    }
  }
}, 30000);
```

## Load Balancing Multiple Gateways

```typescript
// Primary gateway
const primary = new Gateway({
  port: 9999,
  servers: [/* ... */],
});

// Backup gateway
const backup = new Gateway({
  port: 9998,
  servers: [/* ... */],
});

await Promise.all([
  primary.start(),
  backup.start(),
]);

// Health check endpoint
import { createServer } from 'http';

createServer((req, res) => {
  if (req.url === '/health') {
    const status = {
      primary: primary.isRunning(),
      backup: backup.isRunning(),
      primaryServers: primary.listServers().length,
      backupServers: backup.listServers().length,
    };
    res.writeHead(200, { 'Content-Type': 'application/json' });
    res.end(JSON.stringify(status));
  } else {
    res.writeHead(404);
    res.end();
  }
}).listen(8080);
```
