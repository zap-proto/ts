A complete example of an AI agent that discovers and uses tools via ZAP.

## Full Implementation

```typescript
import { Client, Tool, ZapError, ToolNotFoundError } from '@hanzo/zap';

interface Message {
  role: 'user' | 'assistant' | 'system' | 'tool';
  content: string;
  toolCallId?: string;
  toolName?: string;
}

interface ToolCall {
  id: string;
  name: string;
  arguments: Record<string, unknown>;
}

class ZapAgent {
  private client: Client | null = null;
  private tools: Tool[] = [];
  private messages: Message[] = [];

  constructor(private systemPrompt: string = 'You are a helpful assistant.') {
    this.messages.push({ role: 'system', content: systemPrompt });
  }

  async connect(url: string): Promise<void> {
    this.client = await Client.connect(url);
    this.tools = await this.client.listTools();
    console.log(`Connected with ${this.tools.length} tools available`);
  }

  async disconnect(): Promise<void> {
    await this.client?.close();
    this.client = null;
  }

  getToolDescriptions(): string {
    return this.tools
      .map(t => `- ${t.name}: ${t.description}`)
      .join('\n');
  }

  async executeToolCall(call: ToolCall): Promise<string> {
    if (!this.client) {
      throw new ZapError('Not connected');
    }

    try {
      const result = await this.client.callTool(call.name, call.arguments);
      return JSON.stringify(result);
    } catch (error) {
      if (error instanceof ToolNotFoundError) {
        return `Error: Tool '${call.name}' not found`;
      }
      throw error;
    }
  }

  async chat(userMessage: string): Promise<string> {
    this.messages.push({ role: 'user', content: userMessage });

    // In real implementation, send to LLM and parse tool calls
    // This is a simplified example

    // Check if message mentions a tool
    const toolMention = this.tools.find(t =>
      userMessage.toLowerCase().includes(t.name.toLowerCase())
    );

    if (toolMention) {
      // Simulate tool call
      const call: ToolCall = {
        id: crypto.randomUUID(),
        name: toolMention.name,
        arguments: this.parseArguments(userMessage),
      };

      const result = await this.executeToolCall(call);

      this.messages.push({
        role: 'tool',
        content: result,
        toolCallId: call.id,
        toolName: call.name,
      });

      // Generate response based on tool result
      const response = `I used the ${call.name} tool and got: ${result}`;
      this.messages.push({ role: 'assistant', content: response });

      return response;
    }

    // No tool needed - simple response
    const response = `I understood: "${userMessage}"`;
    this.messages.push({ role: 'assistant', content: response });

    return response;
  }

  private parseArguments(message: string): Record<string, unknown> {
    // Simple argument extraction - in production, use LLM
    const args: Record<string, unknown> = {};

    // Look for quoted strings
    const matches = message.match(/"([^"]+)"/g);
    if (matches && matches.length > 0) {
      args.query = matches[0].replace(/"/g, '');
    }

    return args;
  }
}

// Usage
async function main() {
  const agent = new ZapAgent(
    'You are an AI assistant with access to various tools. Use them to help the user.'
  );

  try {
    await agent.connect('zap://localhost:9999');

    console.log('Available tools:');
    console.log(agent.getToolDescriptions());

    // Chat loop
    const response = await agent.chat('Can you search for "TypeScript best practices"?');
    console.log('Agent:', response);

  } finally {
    await agent.disconnect();
  }
}

main().catch(console.error);
```

## With OpenAI Integration

```typescript
import OpenAI from 'openai';
import { Client, Tool } from '@hanzo/zap';

const openai = new OpenAI();

function toolToOpenAI(tool: Tool): OpenAI.ChatCompletionTool {
  return {
    type: 'function',
    function: {
      name: tool.name,
      description: tool.description,
      parameters: tool.schema as OpenAI.FunctionParameters,
    },
  };
}

async function runAgent(userMessage: string) {
  const client = await Client.connect('zap://localhost:9999');
  const tools = await client.listTools();

  const messages: OpenAI.ChatCompletionMessageParam[] = [
    { role: 'system', content: 'You are a helpful assistant with access to tools.' },
    { role: 'user', content: userMessage },
  ];

  const response = await openai.chat.completions.create({
    model: 'gpt-4o',
    messages,
    tools: tools.map(toolToOpenAI),
  });

  const message = response.choices[0]?.message;

  if (message?.tool_calls) {
    for (const call of message.tool_calls) {
      const args = JSON.parse(call.function.arguments);
      const result = await client.callTool(call.function.name, args);

      messages.push(message);
      messages.push({
        role: 'tool',
        tool_call_id: call.id,
        content: JSON.stringify(result),
      });
    }

    // Get final response
    const finalResponse = await openai.chat.completions.create({
      model: 'gpt-4o',
      messages,
    });

    console.log(finalResponse.choices[0]?.message.content);
  } else {
    console.log(message?.content);
  }

  await client.close();
}

runAgent('Search for information about Rust programming language');
```

## With Streaming

```typescript
import { Client } from '@hanzo/zap';

async function streamingAgent(userMessage: string) {
  const client = await Client.connect('zap://localhost:9999');

  // Get tools for context
  const tools = await client.listTools();

  console.log(`Processing: ${userMessage}`);
  console.log(`Available tools: ${tools.map(t => t.name).join(', ')}`);

  // Execute relevant tools in parallel
  const toolResults = await Promise.all(
    tools
      .filter(t => userMessage.toLowerCase().includes(t.name.split('_')[0]!))
      .map(async t => {
        const result = await client.callTool(t.name, { query: userMessage });
        return { tool: t.name, result };
      })
  );

  for (const { tool, result } of toolResults) {
    console.log(`\n[${tool}]:`, result);
  }

  await client.close();
}
```
