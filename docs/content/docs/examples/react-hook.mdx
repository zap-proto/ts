## useZapClient Hook

```tsx
import { useState, useEffect, useCallback, useRef } from 'react';
import { Client, Tool, Resource, ZapError } from '@hanzo/zap';

interface UseZapClientOptions {
  autoConnect?: boolean;
  reconnect?: boolean;
  reconnectDelay?: number;
}

interface UseZapClientReturn {
  client: Client | null;
  tools: Tool[];
  resources: Resource[];
  connected: boolean;
  loading: boolean;
  error: ZapError | null;
  connect: () => Promise<void>;
  disconnect: () => Promise<void>;
}

export function useZapClient(
  url: string,
  options: UseZapClientOptions = {}
): UseZapClientReturn {
  const {
    autoConnect = true,
    reconnect = true,
    reconnectDelay = 5000,
  } = options;

  const [client, setClient] = useState<Client | null>(null);
  const [tools, setTools] = useState<Tool[]>([]);
  const [resources, setResources] = useState<Resource[]>([]);
  const [connected, setConnected] = useState(false);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<ZapError | null>(null);

  const reconnectTimeoutRef = useRef<NodeJS.Timeout>();
  const mountedRef = useRef(true);

  const connect = useCallback(async () => {
    if (!mountedRef.current) return;

    setLoading(true);
    setError(null);

    try {
      const zapClient = await Client.connect(url);

      if (!mountedRef.current) {
        await zapClient.close();
        return;
      }

      const [fetchedTools, fetchedResources] = await Promise.all([
        zapClient.listTools(),
        zapClient.listResources(),
      ]);

      if (!mountedRef.current) {
        await zapClient.close();
        return;
      }

      setClient(zapClient);
      setTools(fetchedTools);
      setResources(fetchedResources);
      setConnected(true);
    } catch (err) {
      if (mountedRef.current) {
        setError(err as ZapError);

        if (reconnect) {
          reconnectTimeoutRef.current = setTimeout(() => {
            if (mountedRef.current) {
              connect();
            }
          }, reconnectDelay);
        }
      }
    } finally {
      if (mountedRef.current) {
        setLoading(false);
      }
    }
  }, [url, reconnect, reconnectDelay]);

  const disconnect = useCallback(async () => {
    if (reconnectTimeoutRef.current) {
      clearTimeout(reconnectTimeoutRef.current);
    }

    if (client) {
      await client.close();
      setClient(null);
      setConnected(false);
      setTools([]);
      setResources([]);
    }
  }, [client]);

  useEffect(() => {
    mountedRef.current = true;

    if (autoConnect) {
      connect();
    }

    return () => {
      mountedRef.current = false;
      if (reconnectTimeoutRef.current) {
        clearTimeout(reconnectTimeoutRef.current);
      }
      client?.close();
    };
  }, [url]); // Only reconnect if URL changes

  return {
    client,
    tools,
    resources,
    connected,
    loading,
    error,
    connect,
    disconnect,
  };
}
```

## useToolCall Hook

```tsx
import { useState, useCallback } from 'react';
import { Client, ZapError } from '@hanzo/zap';

interface UseToolCallOptions {
  onSuccess?: (result: unknown) => void;
  onError?: (error: ZapError) => void;
}

interface UseToolCallReturn<T = unknown> {
  execute: (name: string, args: Record<string, unknown>) => Promise<T | undefined>;
  result: T | null;
  loading: boolean;
  error: ZapError | null;
  reset: () => void;
}

export function useToolCall<T = unknown>(
  client: Client | null,
  options: UseToolCallOptions = {}
): UseToolCallReturn<T> {
  const { onSuccess, onError } = options;

  const [result, setResult] = useState<T | null>(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<ZapError | null>(null);

  const execute = useCallback(async (
    name: string,
    args: Record<string, unknown>
  ): Promise<T | undefined> => {
    if (!client) {
      const err = new ZapError('Client not connected');
      setError(err);
      onError?.(err);
      return;
    }

    setLoading(true);
    setError(null);

    try {
      const output = await client.callTool(name, args) as T;
      setResult(output);
      onSuccess?.(output);
      return output;
    } catch (err) {
      const zapError = err as ZapError;
      setError(zapError);
      onError?.(zapError);
    } finally {
      setLoading(false);
    }
  }, [client, onSuccess, onError]);

  const reset = useCallback(() => {
    setResult(null);
    setError(null);
    setLoading(false);
  }, []);

  return { execute, result, loading, error, reset };
}
```

## useResource Hook

```tsx
import { useState, useEffect, useCallback } from 'react';
import { Client, ResourceContent, ZapError } from '@hanzo/zap';

interface UseResourceOptions {
  autoFetch?: boolean;
  refetchInterval?: number;
}

interface UseResourceReturn {
  content: ResourceContent | null;
  loading: boolean;
  error: ZapError | null;
  refetch: () => Promise<void>;
}

export function useResource(
  client: Client | null,
  uri: string,
  options: UseResourceOptions = {}
): UseResourceReturn {
  const { autoFetch = true, refetchInterval } = options;

  const [content, setContent] = useState<ResourceContent | null>(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<ZapError | null>(null);

  const refetch = useCallback(async () => {
    if (!client || !uri) return;

    setLoading(true);
    setError(null);

    try {
      const data = await client.readResource(uri);
      setContent(data);
    } catch (err) {
      setError(err as ZapError);
    } finally {
      setLoading(false);
    }
  }, [client, uri]);

  useEffect(() => {
    if (autoFetch && client && uri) {
      refetch();
    }
  }, [client, uri, autoFetch]);

  useEffect(() => {
    if (refetchInterval && client && uri) {
      const interval = setInterval(refetch, refetchInterval);
      return () => clearInterval(interval);
    }
  }, [refetchInterval, client, uri, refetch]);

  return { content, loading, error, refetch };
}
```

## Complete Example App

```tsx
import React, { useState } from 'react';
import { useZapClient, useToolCall } from './hooks';

function App() {
  const {
    client,
    tools,
    connected,
    loading: connecting,
    error: connectionError,
  } = useZapClient('wss://gateway.example.com:9999');

  const [query, setQuery] = useState('');

  const {
    execute: search,
    result: searchResult,
    loading: searching,
    error: searchError,
  } = useToolCall(client);

  const handleSearch = async (e: React.FormEvent) => {
    e.preventDefault();
    if (query.trim()) {
      await search('search', { query: query.trim() });
    }
  };

  if (connecting) {
    return <div className="p-4">Connecting to ZAP gateway...</div>;
  }

  if (connectionError) {
    return (
      <div className="p-4 text-red-600">
        Connection error: {connectionError.message}
      </div>
    );
  }

  return (
    <div className="p-4 max-w-2xl mx-auto">
      <h1 className="text-2xl font-bold mb-4">ZAP Agent</h1>

      <div className="mb-4 p-2 bg-green-100 rounded">
        {connected ? 'Connected' : 'Disconnected'} - {tools.length} tools available
      </div>

      <form onSubmit={handleSearch} className="mb-4">
        <input
          type="text"
          value={query}
          onChange={(e) => setQuery(e.target.value)}
          placeholder="Enter search query..."
          className="border p-2 w-full rounded"
          disabled={searching}
        />
        <button
          type="submit"
          disabled={searching || !query.trim()}
          className="mt-2 px-4 py-2 bg-blue-500 text-white rounded disabled:opacity-50"
        >
          {searching ? 'Searching...' : 'Search'}
        </button>
      </form>

      {searchError && (
        <div className="p-2 bg-red-100 text-red-800 rounded mb-4">
          Error: {searchError.message}
        </div>
      )}

      {searchResult && (
        <div className="p-4 bg-gray-100 rounded">
          <h2 className="font-bold mb-2">Results:</h2>
          <pre className="whitespace-pre-wrap text-sm">
            {JSON.stringify(searchResult, null, 2)}
          </pre>
        </div>
      )}

      <div className="mt-8">
        <h2 className="font-bold mb-2">Available Tools:</h2>
        <ul className="space-y-2">
          {tools.map((tool) => (
            <li key={tool.name} className="p-2 border rounded">
              <strong>{tool.name}</strong>
              <p className="text-sm text-gray-600">{tool.description}</p>
            </li>
          ))}
        </ul>
      </div>
    </div>
  );
}

export default App;
```

## Context Provider

```tsx
import React, { createContext, useContext, ReactNode } from 'react';
import { Client, Tool, Resource, ZapError } from '@hanzo/zap';
import { useZapClient } from './hooks';

interface ZapContextValue {
  client: Client | null;
  tools: Tool[];
  resources: Resource[];
  connected: boolean;
  loading: boolean;
  error: ZapError | null;
}

const ZapContext = createContext<ZapContextValue | null>(null);

export function ZapProvider({
  url,
  children,
}: {
  url: string;
  children: ReactNode;
}) {
  const value = useZapClient(url);
  return <ZapContext.Provider value={value}>{children}</ZapContext.Provider>;
}

export function useZap(): ZapContextValue {
  const context = useContext(ZapContext);
  if (!context) {
    throw new Error('useZap must be used within a ZapProvider');
  }
  return context;
}

// Usage
function MyComponent() {
  const { client, tools, connected } = useZap();
  // ...
}
```
