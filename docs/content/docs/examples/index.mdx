## Minimal Client

```typescript
import { Client } from '@hanzo/zap';

const client = await Client.connect('zap://localhost:9999');
const tools = await client.listTools();
console.log('Tools:', tools.map(t => t.name).join(', '));
await client.close();
```

## Minimal Server

```typescript
import { Server } from '@hanzo/zap';

const server = new Server({ port: 9999 });
await server.run();
```

## Minimal Gateway

```typescript
import { Gateway } from '@hanzo/zap';

const gateway = new Gateway({
  port: 9999,
  servers: [
    {
      name: 'fs',
      url: 'stdio://fs',
      transport: 'stdio',
      command: 'npx',
      args: ['-y', '@modelcontextprotocol/server-filesystem', '/tmp'],
    },
  ],
});

await gateway.start();
```
