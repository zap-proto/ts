---
title: Gateway
description: Gateway class for aggregating multiple MCP servers
---

The `Gateway` class aggregates multiple MCP servers behind a single endpoint, providing unified access to tools and resources.

## Import

```typescript
import { Gateway } from '@hanzo/zap';
```

## Constructor

### `new Gateway(config?: Partial<Config>)`

Create a new gateway with optional configuration.

```typescript
const gateway = new Gateway({
  listen: '0.0.0.0',
  port: 9999,
  servers: [
    {
      name: 'filesystem',
      url: 'stdio://mcp-server-filesystem',
      transport: 'stdio',
      command: 'npx',
      args: ['-y', '@modelcontextprotocol/server-filesystem', '/data'],
    },
  ],
});
```

## Server Configuration

```typescript
interface ServerConfig {
  name: string;           // Unique server identifier
  url: string;            // Server URL
  transport: Transport;   // Transport type
  command?: string;       // Command for stdio transport
  args?: string[];        // Command arguments
  env?: Record<string, string>; // Environment variables
}

type Transport = 'stdio' | 'http' | 'websocket' | 'zap' | 'unix';
```

## Methods

### `start(): Promise<void>`

Start the gateway and connect to configured servers.

```typescript
await gateway.start();
console.log('Gateway running');
```

**Throws:** `ZapError` if gateway is already running

### `stop(): Promise<void>`

Stop the gateway and disconnect all servers.

```typescript
await gateway.stop();
```

### `connectServer(config: ServerConfig): Promise<string>`

Connect to an additional server at runtime.

```typescript
const serverId = await gateway.connectServer({
  name: 'new-server',
  url: 'http://localhost:3001',
  transport: 'http',
});

console.log('Connected server:', serverId);
```

**Returns:** `Promise<string>` - Server ID (UUID)

### `disconnectServer(id: string): Promise<void>`

Disconnect a server by ID.

```typescript
await gateway.disconnectServer(serverId);
```

**Throws:** `ZapError` if server not found

### `listServers(): ConnectedServer[]`

List all connected servers.

```typescript
const servers = gateway.listServers();

for (const server of servers) {
  console.log(`${server.name} (${server.status})`);
  console.log(`  Tools: ${server.tools}, Resources: ${server.resources}`);
}
```

### `getServer(id: string): ConnectedServer | undefined`

Get a specific server by ID.

```typescript
const server = gateway.getServer(serverId);
if (server && server.status === 'connected') {
  console.log('Server ready:', server.name);
}
```

### `listTools(): Promise<Tool[]>`

List all tools from all connected servers.

```typescript
const tools = await gateway.listTools();
console.log('Total tools:', tools.length);
```

### `callTool(serverId: string, name: string, args: Record<string, unknown>): Promise<unknown>`

Call a tool on a specific server.

```typescript
const result = await gateway.callTool(serverId, 'read_file', {
  path: '/tmp/example.txt',
});
```

### `listResources(): Promise<Resource[]>`

List all resources from all connected servers.

```typescript
const resources = await gateway.listResources();
```

### `readResource(serverId: string, uri: string): Promise<unknown>`

Read a resource from a specific server.

```typescript
const content = await gateway.readResource(serverId, 'file:///tmp/doc.md');
```

### `isRunning(): boolean`

Check if the gateway is running.

```typescript
if (gateway.isRunning()) {
  console.log('Gateway is active');
}
```

### `getConfig(): Config`

Get a copy of the gateway configuration.

```typescript
const config = gateway.getConfig();
console.log('Port:', config.port);
```

## Example: Multi-Server Gateway

```typescript
import { Gateway, Config } from '@hanzo/zap';

const config: Partial<Config> = {
  listen: '0.0.0.0',
  port: 9999,
  logLevel: 'info',
  servers: [
    // Filesystem access via stdio
    {
      name: 'filesystem',
      url: 'stdio://filesystem',
      transport: 'stdio',
      command: 'npx',
      args: ['-y', '@modelcontextprotocol/server-filesystem', '/home/user'],
    },
    // Web fetching via HTTP
    {
      name: 'fetch',
      url: 'http://localhost:3001',
      transport: 'http',
    },
    // Database access via Unix socket
    {
      name: 'database',
      url: 'unix:///var/run/mcp-db.sock',
      transport: 'unix',
    },
  ],
};

const gateway = new Gateway(config);

// Start gateway
await gateway.start();
console.log(`Gateway ready with ${gateway.listServers().length} servers`);

// Handle shutdown
process.on('SIGINT', async () => {
  await gateway.stop();
  process.exit(0);
});

// Keep running
await new Promise(() => {});
```

## ConnectedServer Type

```typescript
interface ConnectedServer {
  id: string;               // Unique server ID
  name: string;             // Server name from config
  url: string;              // Server URL
  status: ServerStatus;     // Current connection status
  tools: number;            // Number of available tools
  resources: number;        // Number of available resources
}

type ServerStatus = 'connecting' | 'connected' | 'disconnected' | 'error';
```

## Dynamic Server Management

```typescript
// Add server at runtime
const id = await gateway.connectServer({
  name: 'dynamic-server',
  url: 'wss://api.example.com/mcp',
  transport: 'websocket',
});

// Monitor status
const server = gateway.getServer(id);
console.log('Status:', server?.status);

// Remove when done
await gateway.disconnectServer(id);
```
